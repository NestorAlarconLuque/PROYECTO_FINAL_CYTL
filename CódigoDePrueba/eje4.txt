CLASE Calculadora {
    ENTERO sumar(ENTERO a, ENTERO b) {
        ENTERO resultado = a + b$;        
        RETORNAR resultado;
    }
    REAL dividir(REAL x, REAL y) {
        SI (y != 0) {
            @IMPRIMIR "División válida"; 
            RETORNAR x / y;
        } SINO {
            RETORNAR 0.0;
        }
    }

    }

    }
